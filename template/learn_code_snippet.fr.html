(* Si [l] est la liste [e1;…;eN] *)
(* Additionner tous les éléments: *)
List.fold_left ( +. ) 0. l

(* Multiplier tous les éléments: *)
List.fold_left ( *. ) 1. l

(* Conjonction sur une liste de booléens: *)
List.fold_left ( && ) true l

(* Inverser la liste : [eN;…;e1] *)
List.fold_left  (fun r e -> e :: r) [] l
